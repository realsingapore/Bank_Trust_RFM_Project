# 1. Navigate to project folder
cd C:\Users\HP\Desktop\Bank_Trust_RFM

# (Optional) Remove any previous Git history
rmdir /s /q .git

# 2. Create .gitignore file (then add lines like: Data/ and .env inside it)
# On Windows (PowerShell):
ni .gitignore

# 3. Configure git
git config --global user.name "realsingapore"
git config --global user.email "realsingapore@gmail.com"

# 4. Initialize repository
git init

# 5. Add and commit all files
git add .
git commit -m "Initial commit: Bank Trust RFM Project"

# 6. Add remote repository
git remote add origin repo-link (/realsingapore/Bank_Trust_RFM_Project.git)

# 7. Rename branch to 'main' and push to GitHub
git branch -M main
git push -u origin main

# 8. Check which file changes were made
git status

# 9. Add all changes
git add .

# 10. Commit the changes
git commit -m "Add new feature: description of changes"

# 11. Push to GitHub
git push origin main
